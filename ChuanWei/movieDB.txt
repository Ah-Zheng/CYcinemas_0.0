create database ambassador default character set utf8;

create table movies
(
  id int not null auto_increment primary key,
  movieName varchar(20),
  movieId varchar(50),
  rating varchar(20),
  runTime varchar(10),
  info text,
  actor varchar(20),
  genre varchar(10),
  playDate varchar(20),
  poster text
);

create table theaters
(
 id int not null auto_increment primary key,
 theaterName varchar(20),
 address varchar(30),
 phone varchar(20),
 img text
);

create table movietime
(
  id int not null auto_increment primary key,
  movieId varchar(50),
  theaterName varchar(20),
  seatTag varchar(20),
  time varchar(10),
  seatInfo varchar(10)
);

create table movieday
(
  id int not null auto_increment primary key,
  movieId varchar(50),
  weekday varchar(20),
  date date
);
